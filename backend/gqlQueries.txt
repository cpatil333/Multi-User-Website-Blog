Queries
===========
1.Get User List
query{
  users {
    id
    name
    email
    role
  }
}

2. Get the Post List
query{
  posts {
    id
    title
    content
  }
}

3.Get Comments List
query{
  comments {
    id
    text
  }
}

4.Get Post by id
query($postId: ID!){
  post(id: $postId) {
    id
    title
    content
    author {
      name
    }
    comments {
      id
      text
      author {
        name
      }
    }
  }
}
